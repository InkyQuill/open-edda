# Writer Project Core Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build the first working Project Core: a single-deployable Go/React app with SQLite-backed Markdown Story Projects, Chapters, Story Bible Entries, Writing Briefs, Notes, Per-Item Revisions, and Elysium Layout import/export.

**Architecture:** The Go service owns persistence, migrations, import/export, revisions, and the HTTP API. The React frontend uses Vite and TypeScript, calls the Go API, and provides a minimal Project Dashboard plus Writing Workspace shell. Agent Core, provider calls, Skills, Prompt Records, and Direct Apply are intentionally planned as later milestones.

**Tech Stack:** Go 1.26, `chi`, `sqlc`, `goose`, SQLite, React, TypeScript, Vite, Node 26, latest Bun, mise.

---

## Scope Boundary

This plan implements Milestone 1 from the approved spec: Project Core. It does not implement model providers, agent sessions, Skills, Direct Apply, Activity Trails, Prompt Records, or mobile polish. Those need separate plans after Project Core is working.

```go
package app

import (
	"encoding/json"
	"net/http"

	"github.com/go-chi/chi/v5"
)

type Dependencies struct{}

func New(deps *Dependencies) http.Handler {
	r := chi.NewRouter()
	r.Get("/api/health", func(w http.ResponseWriter, r *http.Request) {
		writeJSON(w, http.StatusOK, map[string]string{"status": "ok"})
	})
	return r
}

func writeJSON(w http.ResponseWriter, status int, value any) {
	w.Header().Set("Content-Type", "application/json")
	w.WriteHeader(status)
	_ = json.NewEncoder(w).Encode(value)
}
```

```go
package store

import (
	"database/sql"
	"fmt"

	_ "github.com/mattn/go-sqlite3"
)

func Open(path string) (*sql.DB, error) {
	db, err := sql.Open("sqlite3", path)
	if err != nil {
		return nil, fmt.Errorf("open sqlite: %w", err)
	}

	pragmas := []string{
		"PRAGMA foreign_keys = ON",
		"PRAGMA journal_mode = WAL",
		"PRAGMA busy_timeout = 5000",
	}
	for _, pragma := range pragmas {
		if _, err := db.Exec(pragma); err != nil {
			_ = db.Close()
			return nil, fmt.Errorf("apply %s: %w", pragma, err)
		}
	}

	return db, nil
}
```

```go
package project

import (
	"context"
	"database/sql"
	"errors"
	"fmt"
	"strings"
	"time"

	"git.inkyquill.net/inky/writer/store"
)

var ErrConflict = errors.New("content revision conflict")

type Service struct {
	queries *store.Queries
}

func NewService(queries *store.Queries) *Service {
	return &Service{queries: queries}
}

func (s *Service) CreateContent(ctx context.Context, input CreateContentInput) (ContentItem, error) {
	now := time.Now().UTC().Format(time.RFC3339)
	item := ContentItem{
		ID:              newID(),
		ProjectID:       input.ProjectID,
		Kind:            input.Kind,
		Title:           input.Title,
		Slug:            slugify(input.Title),
		BodyMarkdown:    input.BodyMarkdown,
		MetadataJSON:    defaultJSON(input.MetadataJSON),
		SortOrder:       input.SortOrder,
		CurrentRevision: 1,
	}

	err := s.queries.CreateContentItem(ctx, store.CreateContentItemParams{
		ID:              item.ID,
		ProjectID:       item.ProjectID,
		Kind:            string(item.Kind),
		Title:           item.Title,
		Slug:            item.Slug,
		BodyMarkdown:    item.BodyMarkdown,
		MetadataJson:    item.MetadataJSON,
		SortOrder:       item.SortOrder,
		CurrentRevision: item.CurrentRevision,
		CreatedAt:       now,
		UpdatedAt:       now,
	})
	if err != nil {
		return ContentItem{}, fmt.Errorf("create content item: %w", err)
	}

	if err := s.queries.CreateRevision(ctx, store.CreateRevisionParams{
		ID:             newID(),
		ContentItemID:  item.ID,
		RevisionNumber: item.CurrentRevision,
		BodyMarkdown:   item.BodyMarkdown,
		MetadataJson:   item.MetadataJSON,
		Reason:         emptyDefault(input.Reason, "create"),
		CreatedBy:      emptyDefault(input.CreatedBy, "author"),
		CreatedAt:      now,
	}); err != nil {
		return ContentItem{}, fmt.Errorf("create initial revision: %w", err)
	}

	return item, nil
}

func (s *Service) UpdateContent(ctx context.Context, input UpdateContentInput) (ContentItem, error) {
	current, err := s.queries.GetContentItem(ctx, store.GetContentItemParams{
		ID:        input.ContentID,
		ProjectID: input.ProjectID,
	})
	if err != nil {
		return ContentItem{}, fmt.Errorf("get content item: %w", err)
	}
	if current.CurrentRevision != input.ExpectedRevision {
		return ContentItem{}, ErrConflict
	}

	nextRevision := current.CurrentRevision + 1
	now := time.Now().UTC().Format(time.RFC3339)
	metadata := defaultJSON(input.MetadataJSON)

	rows, err := s.queries.UpdateContentItemBody(ctx, store.UpdateContentItemBodyParams{
		BodyMarkdown:      input.BodyMarkdown,
		MetadataJson:      metadata,
		CurrentRevision:   nextRevision,
		UpdatedAt:         now,
		ID:                input.ContentID,
		ProjectID:         input.ProjectID,
		CurrentRevision_2: input.ExpectedRevision,
	})
	if err != nil {
		return ContentItem{}, fmt.Errorf("update content item: %w", err)
	}
	if rows == 0 {
		return ContentItem{}, ErrConflict
	}

	if err := s.queries.CreateRevision(ctx, store.CreateRevisionParams{
		ID:             newID(),
		ContentItemID:  input.ContentID,
		RevisionNumber: nextRevision,
		BodyMarkdown:   input.BodyMarkdown,
		MetadataJson:   metadata,
		Reason:         emptyDefault(input.Reason, "update"),
		CreatedBy:      emptyDefault(input.CreatedBy, "author"),
		CreatedAt:      now,
	}); err != nil {
		return ContentItem{}, fmt.Errorf("create revision: %w", err)
	}

	return ContentItem{
		ID:              current.ID,
		ProjectID:       current.ProjectID,
		Kind:            ContentKind(current.Kind),
		Title:           current.Title,
		Slug:            current.Slug,
		BodyMarkdown:    input.BodyMarkdown,
		MetadataJSON:    metadata,
		SortOrder:       current.SortOrder,
		CurrentRevision: nextRevision,
	}, nil
}

func defaultJSON(value string) string {
	if strings.TrimSpace(value) == "" {
		return "{}"
	}
	return value
}

func emptyDefault(value, fallback string) string {
	if strings.TrimSpace(value) == "" {
		return fallback
	}
	return value
}

func slugify(value string) string {
	value = strings.ToLower(strings.TrimSpace(value))
	value = strings.ReplaceAll(value, " ", "-")
	if value == "" {
		return "untitled"
	}
	return value
}

func newID() string {
	return strings.ReplaceAll(time.Now().UTC().Format("20060102150405.000000000"), ".", "")
}
```
